var isAnagram = function(s, t) {
    if(s.length !== t.length){
        return false;
    }
    else{
        let s1 = s.split("").sort().reverse().join("");
        let t1 = t.split("").sort().reverse().join("");
        for(let i=0;i<s1.length;i++){
            if(s1[i] !== t1[i]){
                return false;
            }
        }
        return true;
    }
};
