var lengthOfLongestSubstring = function(s) {
    let sub = '';
    let maxLen = 0;

    if (s.length === 0) {
        return 0;
    }
    for (let i = 0; i < s.length; i++) {
        if (sub.includes(s[i])) {
            sub = sub.slice(sub.indexOf(s[i]) + 1);
        }
        sub += s[i];
        if (sub.length > maxLen) {
            maxLen = sub.length;
        }
    }

    return maxLen;
};
